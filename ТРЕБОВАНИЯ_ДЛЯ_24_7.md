### 4. 🛡️ **БЕЗОПАСНОСТЬ И НАДЕЖНОСТЬ**

#### A. Rate Limiting
```python
from aiogram.utils.token import validate_token
from aiogram.exceptions import TelegramRetryAfter
import asyncio

async def rate_limited_request(func, *args, **kwargs):
    """Обработка rate limit"""
    try:
        return await func(*args, **kwargs)
    except TelegramRetryAfter as e:
        await asyncio.sleep(e.retry_after)
        return await func(*args, **kwargs)
```

#### B. Graceful Shutdown
```python
import signal
import sys

def signal_handler(signum, frame):
    """Обработка сигналов для graceful shutdown"""
    global bot_running
    logger.info(f"Получен сигнал {signum}, завершаем работу...")
    bot_running = False
    sys.exit(0)

signal.signal(signal.SIGINT, signal_handler)
signal.signal(signal.SIGTERM, signal_handler)
```

### 6. 🚨 **АЛЕРТИНГ И УВЕДОМЛЕНИЯ**

#### A. Webhook уведомления
```python
import aiohttp

async def send_alert(message: str, severity: str = "warning"):
    """Отправка алертов в Slack/Discord"""
    webhook_url = os.getenv('ALERT_WEBHOOK_URL')
    if not webhook_url:
        return
    
    payload = {
        "text": f"🚨 {severity.upper()}: {message}",
        "timestamp": datetime.now().isoformat()
    }
    
    async with aiohttp.ClientSession() as session:
        await session.post(webhook_url, json=payload)
```

#### B. Email уведомления
```python
import smtplib
from email.mime.text import MIMEText

async def send_email_alert(subject: str, body: str):
    """Отправка email алертов"""
    smtp_server = os.getenv('SMTP_SERVER')
    smtp_port = int(os.getenv('SMTP_PORT', 587))
    email_user = os.getenv('EMAIL_USER')
    email_password = os.getenv('EMAIL_PASSWORD')
    
    msg = MIMEText(body)
    msg['Subject'] = subject
    msg['From'] = email_user
    msg['To'] = os.getenv('ADMIN_EMAIL')
    
    with smtplib.SMTP(smtp_server, smtp_port) as server:
        server.starttls()
        server.login(email_user, email_password)
        server.send_message(msg)
```

#### B. Rolling updates
```python
import subprocess
import os

async def update_bot():
    """Обновление без простоя"""
    try:
        # Проверка обновлений
        result = subprocess.run(['git', 'pull'], capture_output=True, text=True)
        
        if "Already up to date" not in result.stdout:
            # Установка зависимостей
            subprocess.run(['pip', 'install', '-r', 'requirements.txt'])
            
            # Перезапуск
            os.execv(sys.executable, ['python'] + sys.argv)
    except Exception as e:
        logger.error(f"Ошибка обновления: {e}")
```
